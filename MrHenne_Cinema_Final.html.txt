<script>
let heartbeat = document.getElementById("heartbeat");
let music = document.getElementById("music");

function startFilm(){
    document.getElementById("intro").style.opacity=0;
    document.querySelector("button").style.display="none";

    heartbeat.play();

    setTimeout(()=>{
        music.volume = 0;
        music.play();
        fadeInMusic();
    },4000);

    playScene("scene1","text1",4000);
    playScene("scene2","text2",11000);
    playScene("scene3","text3",18000);
    playScene("scene4","text4",25000);
    playScene("scene5","text5",32000);

    setTimeout(()=>{
        fadeOutScenes();
        document.getElementById("final").style.opacity=1;
    },39000);
}

function playScene(sceneId,textId,delay){
    setTimeout(()=>{
        clearScenes();
        let scene=document.getElementById(sceneId);
        let text=document.getElementById(textId);
        scene.style.opacity=1;
        text.style.opacity=1;
    },delay);
}

function clearScenes(){
    let scenes=document.querySelectorAll(".scene");
    scenes.forEach(s=>s.style.opacity=0);
}

function fadeInMusic(){
    let fade = setInterval(()=>{
        if(music.volume < 1){
            music.volume += 0.05;
        } else {
            clearInterval(fade);
        }
    },300);
}

function fadeOutScenes(){
    let scenes=document.querySelectorAll(".scene");
    scenes.forEach(s=>s.style.opacity=0);
}
